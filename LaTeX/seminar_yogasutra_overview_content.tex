\begin{frame}[fragile]\frametitle{Why Read the Yoga Sutras?}

\begin{center}
{\large ``If you are on any path where you want to be happy, to be free of suffering, and understand the mind and consciousness, the Sutras are a must-read.''}\\
, \textbf{Prof. Edwin Bryant}, Rutgers University
\end{center}

\vspace{3mm}
\begin{itemize}
\item Written by Sri Patanjali $\approx$ 2000 years ago:the first step-by-step manual for the \emph{practice} of Yoga
\item Most translated ancient Indian text in the medieval era:40+ Indian languages
\item Based on three principles (Nancy Wile):
    \begin{itemize}
    \item Suffering is not caused by events but by our \emph{thoughts about} events
    \item The unlimited peace we seek is our true identity:it exists within us
    \item Peace and Self-realization are attained by \emph{mastering the mind}
    \end{itemize}
\end{itemize}

\tiny{(Ref: Prof. Edwin Bryant; Nancy Wile, Yoga Education Institute; Wikipedia:Yoga Sutras of Patanjali)}

\begin{frame}[fragile]\frametitle{Three Core Principles:The Foundation}

\begin{center}
{\normalsize \textbf{Nancy Wile, Yoga Education Institute}}
\end{center}

\vspace{4mm}
\begin{enumerate}
\item \textbf{Suffering is Internal}\\
{\small Suffering is not caused by forces outside us, but by our faulty and limited perception of life and of who we are. Suffering is not caused by the \emph{situation}:but by our \emph{thoughts about} the situation.}

\vspace{3mm}
\item \textbf{Peace is Our True Nature}\\
{\small The unwavering peace we seek is realized by experiencing the unlimited and eternal peace that is our true identity. Though hidden by ignorance, it exists within us, waiting to be revealed.}

\vspace{3mm}
\item \textbf{Mastery of Mind = Self-Realization}\\
{\small Peace and self-realization are attained by mastering the mind. Only a single-pointed, calm mind can reveal the true Self.}
\end{enumerate}

\vspace{2mm}
\begin{center}
{\small These three principles underpin all 196 sutras.}
\end{center}

\tiny{(Ref: Nancy Wile, Yoga Education Institute, 2014)}

\end{frame}

\begin{frame}[fragile]\frametitle{Key Yogic Concepts}
\begin{itemize}
\item \textbf{पुरुष (Purusha)}:Pure consciousness; the Seer; eternal, unchanging witness
\item \textbf{प्रकृति (Prakriti)}:Primordial nature; composed of three गुण (सत्व, रजस्, तमस्)
\item \textbf{चित्त (Chitta)}:Mind-stuff; four coordinated functions:
    \begin{itemize}
    \item \textbf{मनस्}:sensory receiver; processes perception and directs action
    \item \textbf{बुद्धि}:intellect; faculty of discrimination and decision
    \item \textbf{अहंकार}:ego; the ``I-maker'' that claims ``I think, I act''
    \item \textbf{चित्त} (memory store):repository of संस्कार (impressions)
    \end{itemize}
\item \textbf{वृत्ति (Vritti)}:modifications/ripples of mind; the disturbance Yoga addresses
\item \textbf{क्लेश (Klesha)}:deep afflictions that generate vrittis and perpetuate suffering
\item \textbf{कैवल्य (Kaivalya)}:liberation; Purusha abiding in its own pure nature
\end{itemize}

\vspace{2mm}
\begin{center}
{\small Chitta is like a river, flowing between worldliness and Kaivalya. (Vyasa)\\
``We learn to control our mind and not be controlled by our thoughts.'':Nancy Wile}
\end{center}

\tiny{(Ref: Vyasa Bhashya; Krishnamurthy Ramakrishnan; Dr. Veena Londhe)}

\end{frame}

\section[Samadhipad]{Chapter 1: Samadhi Pada:Concentration}

\begin{frame}[fragile]\frametitle{}
\begin{center}
{\Large Chapter 1: समाधिपाद Samadhi Pada}\\
{\normalsize Concentration:51 Sutras}\\
{\small For \textbf{uttama} (superior) seekers already near the goal; addresses the nature of mind, the goal of Yoga, and the twin keys of practice and non-attachment.}\\
{\small The first 16 sutras (1.1–1.16) are sometimes called the \textbf{``Sweet Sixteen''}:the complete philosophy of Yoga in a nutshell.}
\end{center}
\tiny{(Ref: Nancy Wile, Yoga Education Institute; Dr. Ananda Balayogi Bhavanani)}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{1.1:अथ योगानुशासनम्}

\begin{sanskrit}
अथ योगानुशासनम् ॥१.१॥
\end{sanskrit}

\begin{itemize}
\item \textbf{अथ}:not a casual ``now''; signals \emph{existential readiness}: ethical maturity, intellectual clarity, earnest longing for liberation
    \begin{itemize}
    \item ``Now, after having done prior preparation through life and other practices, the study and practice of Yoga begins.'' (Nancy Wile)
    \item You are coming from different evolutionary paths; \emph{now} you are eligible
    \end{itemize}
\item \textbf{अनुशासनम्}:systematic transmission; a living tradition being re-organized, not invented; \emph{अनु} = following the existing tradition
\item \textbf{Key insight:} Yoga does not begin with technique or posture:it begins the moment one becomes \emph{ready to be taught}
\end{itemize}

\tiny{(Ref: Krishnamurthy Ramakrishnan; Nancy Wile; Vyasa Bhashya; Prof. Edwin Bryant)}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{1.2:योगश्चित्तवृत्तिनिरोधः}

\begin{sanskrit}
योगश् चित्तवृत्तिनिरोधः ॥१.२॥
\end{sanskrit}

\begin{itemize}
\item The definition of Yoga in four words:not postures, not breathing, but \emph{stilling of mind-modifications}
\item \textbf{चित्त}: four coordinated voices:मनस् receives → बुद्धि evaluates → चित्त supplies memory → अहंकार asserts ``I am doing this''
\item \textbf{वृत्ति}: ripples on the lake of the mind:the disturbances, not the lake itself
\item \textbf{निरोधः}: restraint, regulation, coordination:not suppression (उच्छेद); root \emph{रुध्} = to check movement
\item \textbf{5 states of Chitta:} मूढ (dull) → क्षिप्त (restless) → विक्षिप्त (partly steady) → एकाग्र (focused) → निरुद्ध (fully restrained)
\item Edwin Bryant: ``Yoga is meditative practices culminating in a state of consciousness free from all modes of active or discursive thought.''
\end{itemize}

\tiny{(Ref: Krishnamurthy Ramakrishnan; [SS] Swami Satchidananda; Prof. Edwin Bryant; Vyasa Bhashya)}

\end{frame}

\begin{frame}[fragile]\frametitle{1.3–1.4:The Seer and the Seen}

\begin{sanskrit}
तदा द्रष्टुः स्वरूपेऽवस्थानम् ॥१.३॥
वृत्तिसारूप्यमितरत्र ॥१.४॥
\end{sanskrit}

\textbf{1.3}: ``Then the Seer abides in Itself, resting in its own True Nature, which is called Self-realization.'' (Nancy Wile)\\
\textbf{1.4}:At all other times, the Seer appears to take the form of the vrittis:it takes on the identity of those thought patterns.

\begin{itemize}
\item \textbf{Core metaphor:} Mind = lake; Purusha = the still bottom; Vrittis = ripples. As long as the lake is disturbed, one cannot see the bottom clearly.
\item Yoga stills the lake so the true depth (Purusha) is revealed.
\item \textbf{Identity question:} ``What identities do you give yourself?'':your job, your role, your story. Are those really \emph{you}? (Nancy Wile)
\item We do not \emph{create} the Seer:we \emph{uncover} what was always there.
\end{itemize}

\tiny{(Ref: Nancy Wile; [VH] Vyasa Houston; [BM] Barbara Miller; Swami Vivekananda)}

\end{frame}

\begin{frame}[fragile]\frametitle{1.5–1.6:The Five Vrittis}

\begin{sanskrit}
वृत्तयः पञ्चतय्यः क्लिष्टाक्लिष्टाः ॥१.५॥
प्रमाणविपर्ययविकल्पनिद्रास्मृतयः ॥१.६॥
\end{sanskrit}

\textbf{Five types of modifications:all require restraint:}

\begin{center}
{\small
\begin{tabular}{|p{2cm}|p{2.5cm}|p{2.5cm}|}
\hline
\textbf{Vritti} & \textbf{Meaning} & \textbf{Example} \\
\hline
प्रमाण & Valid knowledge & Direct seeing \\
विपर्यय & Misperception & Rope seen as snake \\
विकल्प & Imagination & ``Rabbit's horn'' \\
निद्रा & Sleep & Absence-based vritti \\
स्मृति & Memory & Retention of impressions \\
\hline
\end{tabular}
}
\end{center}

\begin{itemize}
\item \textbf{क्लिष्ट} (pain-afflicted) vs. \textbf{अक्लिष्ट} (not pain-afflicted): not a moral division; a phenomenological tone
\item Even pleasant vrittis disturb equilibrium:all require restraint
\end{itemize}

\tiny{(Ref: Krishnamurthy Ramakrishnan; Nancy Wile; [IT] I.K. Taimni)}

\end{frame}

%% NEW SLIDE: How a Vritti Forms
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Before the Ripple Surfaces:How a Vritti (वृत्ति) Forms}

\begin{itemize}
\item Chitta (चित्त) holds far more activity than we are ever aware of:most of it stays below the threshold of noticing
\item A Vritti (वृत्ति) begins as an impulse stirring in that depth:not yet a ``thought'' in the way we experience thought
\item It becomes a perceived thought only when attention falls on it:मनस् (Manas) receiving + बुद्धि (Buddhi) evaluating together illuminate a few impulses out of many
\item This is why two people in the same situation can have entirely different thoughts:different impulses get lit up
\item The rising happens largely on its own, prior to conscious choice:we do not decide which impulse surfaces
\end{itemize}

\tiny{(Ref: Swami Vivekananda's lake metaphor; Vritti:Wikipedia; Yoga Sutra 1.2 commentary tradition)}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% NEW SLIDE: Why Nirodha Targets the Rising
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Why Nirodha (निरोध) Targets the Rising, Not the Fight}

\begin{itemize}
\item Since a Vritti (वृत्ति) surfaces before we consciously choose it, fighting it \emph{after} it has risen is already fighting an effect, not the cause
\item Nirodha (निरोध) works upstream:training attention so fewer disturbing impulses get illuminated in the first place
\item Not a suppression of activity (उच्छेद, uchchheda), but a steady redirection:root रुध् (rudh) = to check movement, not to destroy it
\item Memory aid only, not a scriptural distinction:opposing an outside circumstance (loosely, विरोध, virodha) is a different move from this inward, self-directed restraint (निरोध, nirodha)
\item Restraint is carried out through Abhyasa (अभ्यास) and Vairagya (वैराग्य)
\end{itemize}

\tiny{(Ref: root रुध् etymology; Yoga Sutra 1.2 commentary tradition; forward-reference to 1.12)}

\end{frame}

%% NEW SLIDE B1: Colored vs Uncolored Thoughts
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{``Coloring'' Your Thoughts:1.5 in Daily Life}

\begin{center}
{\normalsize \textbf{Many of our thoughts are colored by our judgments and expectations.}}
\end{center}

\begin{itemize}
\item A car is driving down the street = \textbf{fact} (प्रमाण)
\item That car is mine / I want it = \textbf{concept} (colored by desire = राग)
\item Seeing a squirrel run in front of the car, you brake:but it's just dry leaves (विपर्यय)
\end{itemize}

\textbf{Practical practice from Nancy Wile:}
\begin{itemize}
\item When a thought arises, simply note: \emph{``This is colored''} or \emph{``This is not colored''}
\item Ask: \emph{``Is this useful or not useful?''}:observe without fighting or feeding the thought
\item ``These sutras focus on not fighting against anxiety or depression, but just \emph{letting it be} and observing it.'':Nancy Wile
\item Even physical pain: suffering reduces when we observe it with detachment rather than adding a \emph{story} around the pain
\end{itemize}

\begin{center}
{\small The mind may want to make something a problem. It's best to just observe.}
\end{center}

\tiny{(Ref: Nancy Wile, Yoga Education Institute, p.6-7)}

\end{frame}

\begin{frame}[fragile]\frametitle{1.7–1.9:Valid Knowledge, Misperception, Imagination}

\begin{sanskrit}
प्रत्यक्षानुमानागमाः प्रमाणानि ॥१.७॥
विपर्ययो मिथ्याज्ञानमतद्रूपप्रतिष्ठम् ॥१.८॥
शब्दज्ञानानुपाती वस्तुशून्यो विकल्पः ॥१.९॥
\end{sanskrit}

\begin{itemize}
\item \textbf{1.7:Three Pramanas (valid knowledge):}
    \begin{itemize}
    \item \textbf{प्रत्यक्ष}:direct perception
    \item \textbf{अनुमान}:inference (fire from smoke)
    \item \textbf{आगम}:authoritative testimony (reliable teacher or shastra)
    \end{itemize}
\item \textbf{1.8:विपर्यय (Misperception):} false knowledge not grounded in the object's true nature; rope → snake; emotional reactions arise from misperception, not reality
\item \textbf{1.9:विकल्प (Imagination):} word-based cognition without a real object; ``horn of a rabbit''; modern social media massively amplifies vikalpa:we react to images of things that never happened to us
\end{itemize}

\tiny{(Ref: Krishnamurthy Ramakrishnan; Nancy Wile; Patanjaliyogasutra.in)}

\end{frame}

\begin{frame}[fragile]\frametitle{1.10–1.11:Sleep and Memory}

\begin{sanskrit}
अभावप्रत्ययालम्बना वृत्तिर्निद्रा ॥१.१०॥
अनुभूतविषयासंप्रमोषः स्मृतिः ॥१.११॥
\end{sanskrit}

\textbf{1.10:निद्रा (Sleep):}
\begin{itemize}
\item Sleep is not the absence of mind:it is a vritti whose object is \emph{non-existence}
\item ``I slept well'' proves awareness persisted in seed form
\item Deep sleep rehearses Samadhi:the difference: Samadhi brings \emph{awareness} into that stillness
\end{itemize}

\textbf{1.11:स्मृति (Memory):}
\begin{itemize}
\item Memory = \emph{non-loss} of experienced objects; faithful retention of संस्कार (impressions)
\item Tool: enables learning; Trap: fixation on past pain
\item ``We create our own world through our mental projections.'':Nancy Wile
\item Liberation lies \emph{beyond} both memory and forgetting
\end{itemize}

\tiny{(Ref: Krishnamurthy Ramakrishnan; Nancy Wile; [HA] Hariharananda Aranya; [VH] Vyasa Houston)}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{1.12:अभ्यासवैराग्याभ्यां तन्निरोधः}

\begin{sanskrit}
अभ्यासवैराग्याभ्यां तन्निरोधः ॥१.१२॥
\end{sanskrit}

\begin{center}
{\large Restraint of vrittis is achieved through \textbf{Practice} and \textbf{Non-Attachment}}
\end{center}

\begin{columns}
\begin{column}[t]{0.5\linewidth}
\textbf{अभ्यास (Abhyasa):Practice}
\begin{itemize}
\item Choosing and applying effort consistently
\item Brings \emph{steadiness} (sthiti) to the mind
\item The active, disciplined wing
\end{itemize}
\end{column}
\begin{column}[t]{0.5\linewidth}
\textbf{वैराग्य (Vairagya):Non-Attachment}
\begin{itemize}
\item Cultivated dispassion:not indifference
\item Letting go of attachment to results and experiences
\item The releasing, freeing wing
\end{itemize}
\end{column}
\end{columns}

\vspace{3mm}
\begin{center}
{\small Like the two wings of a bird:both are needed for flight.\\
\emph{``Practice and non-attachment are the twin keys upon which the entire system of Yoga rests.''}:Dr. Ananda}
\end{center}

\tiny{(Ref: Dr. Ananda Balayogi Bhavanani; [SS] Swami Satchidananda; Krishnamurthy Ramakrishnan)}

\end{frame}

\begin{frame}[fragile]\frametitle{1.13–1.14:Defining Practice and Making it Firm}

\begin{sanskrit}
तत्र स्थितौ यत्नोऽभ्यासः ॥१.१३॥
स तु दीर्घकालनैरन्तर्यसत्कारासेवितो दृढभूमिः ॥१.१४॥
\end{sanskrit}

\textbf{1.13:Abhyasa defined:} the effort directed toward steadiness (sthiti); choosing actions that bring a stable and tranquil state.

\textbf{1.14:Practice becomes a firm foundation when done with:}
\begin{itemize}
\item \textbf{दीर्घकाल}:for a \emph{long time}; not weeks, but years of sustained commitment
\item \textbf{नैरन्तर्य}:\emph{without break}; continuity; Yoga practice should be steady and without gaps
\item \textbf{सत्कार}:with \emph{earnestness and reverence}; sincere devotion, not mechanical performance
\item \textbf{आसेवित}:faithfully attended to; the whole heart behind the practice
\end{itemize}

\vspace{2mm}
\begin{center}
{\small ``Yoga practice should be steady and without gaps.'':Nancy Wile (1.14)\\
``Yoga is not a shortcut. It is a long road walked with joy.''}
\end{center}

\tiny{(Ref: Nancy Wile; [HA] Hariharananda Aranya; Patanjaliyogasutra.in)}

\end{frame}

\begin{frame}[fragile]\frametitle{1.15–1.16:Vairagya and Para-Vairagya}

\begin{sanskrit}
दृष्टानुश्रविकविषयवितृष्णस्य वशीकारसंज्ञा वैराग्यम् ॥१.१५॥
तत्परं पुरुषख्यातेर्गुणवैतृष्ण्यम् ॥१.१६॥
\end{sanskrit}

\textbf{1.15:Vairagya (Non-attachment):}
\begin{itemize}
\item The mind that has lost craving for objects seen \emph{or} described has achieved vairagya
\item Not hatred of life:cultivated \emph{dispassionate objectivity}
\item ``We give up attachments to things, then to experiences, then to people, then to our own identity.'' (Nancy Wile)
\item Many people identify with labels:job, role, place. Notice what happens when you let go of your story. If they go away, you are still here.
\end{itemize}

\textbf{1.16:Para-Vairagya (Supreme Non-attachment):}
\begin{itemize}
\item Indifference even to the three \textbf{gunas}, achieved through knowledge of \textbf{Purusha}
\item Letting go of even subtle spiritual pleasures and yogic attainments
\item Destroys the very \emph{seeds} of the impurities
\end{itemize}

\tiny{(Ref: Dr. Ananda Balayogi Bhavanani; Nancy Wile; [IT] I.K. Taimni)}

\end{frame}

\begin{frame}[fragile]\frametitle{1.17–1.22:Samadhi Types, Five Powers \& Seeker Intensity}

\textbf{1.17–1.18:Types of Samadhi (Samprajnata / Asamprajnata):}
\begin{itemize}
\item Four levels of object-based absorption: \textbf{Vitarka} (gross) → \textbf{Vichara} (subtle) → \textbf{Ananda} (bliss) → \textbf{Asmita} (I-am-ness):like listening to a song: you first notice it, then examine it, then feel it, then lose yourself in it (Nancy Wile)
\item Asamprajnata (1.18): objectless; only latent impressions remain
\end{itemize}

\textbf{1.20:Five Powers (Panchabala) for the systematic seeker:}
\begin{enumerate}
\item \textbf{श्रद्धा} Shraddha:faith in one's own power to reach enlightenment
\item \textbf{वीर्य} Veerya:strength of body and mind; vigour
\item \textbf{स्मृति} Smriti:repeated memory of the path; keeping the goal alive
\item \textbf{समाधि} Samadhi-prajna:training in deep concentration
\item \textbf{प्रज्ञा} Prajna:pursuit of real knowledge; wisdom
\end{enumerate}

\textbf{1.21–1.22:Intensity matters:}
\begin{itemize}
\item Mild / Medium / Intense practitioners reach the goal at different speeds
\item ``Those who pursue with intensity of feeling, vigour, and firm conviction achieve concentration more quickly.'' (1.21)
\end{itemize}

\tiny{(Ref: Nancy Wile p.8-9; [HA] Hariharananda Aranya; Dr. Ananda Balayogi Bhavanani)}

\end{sanskrit}

\begin{itemize}
\item \textbf{3.38:Critical Warning:} Siddhis are \textbf{attainments} in the waking state but \textbf{obstacles} to Samadhi. ``The temptation of these powers should be avoided and the attention fixed only on liberation.'':Nancy Wile
\end{itemize}

\begin{sanskrit}
तद्वैराग्यादपि दोषबीजक्षये कैवल्यम् ॥३.५१॥
\end{sanskrit}

\begin{itemize}
\item \textbf{3.50–3.51:Para-Vairagya:} Kaivalya is attained only through renunciation even of the highest siddhis. Decline even celestial invitations:anything less than Kaivalya is still bondage.
\item \textbf{3.56:} Liberation comes when Buddhi and Purusha become equal in purity:total discrimination achieved.
\end{itemize}

\tiny{(Ref: Dr. Ananda Balayogi Bhavanani; Nancy Wile; yogasutrastudy.info)}

\end{frame}

%% SECTION: Chapter 4:Kaivalya Pada
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\section[Kaivalyapad]{Chapter 4: Kaivalya Pada:Absolute Freedom}

\begin{frame}[fragile]\frametitle{Chapter 4: कैवल्यपाद Kaivalya Pada:Absolute Freedom (34 Sutras)}

\begin{itemize}
\item \textbf{Theme:} The nature of consciousness, liberation, and what lies beyond all experience
\item Kaivalya = ``aloneness'':not isolation, but the deepest realization where no division between self and other exists
\item ``What happens when you take control of your thoughts rather than having your thoughts control you.'':Nancy Wile
\end{itemize}

\begin{sanskrit}
जन्मौषधिमन्त्रतपःसमाधिजाःसिद्धयः ॥४.१॥
\end{sanskrit}

\begin{itemize}
\item \textbf{4.1:} Siddhis may arise from birth, herbs, mantras, austerities, or Samadhi:but they are never the goal
\item \textbf{4.4–4.6:} All mind-constructs ultimately flow from the ego (asmita); the liberated yogi is free from ego-driven mind
\item \textbf{4.15–4.17:} The same object is perceived differently by different minds:``No wonder everyone seems to have his or her own view of the world.'' (Dr. Ananda)
\end{itemize}

\tiny{(Ref: Dr. Ananda Balayogi Bhavanani; Nancy Wile; Wikipedia)}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{4.7–4.12:Karma and Samskaras}

\begin{sanskrit}
कर्माशुक्लाकृष्णं योगिनस्त्रिविधमितरेषाम् ॥४.७॥
जातिदेशकालव्यवहितानामप्यानन्तर्यं स्मृतिसंस्कारयोरेकरूपत्वात् ॥४.९॥
\end{sanskrit}

\begin{itemize}
\item \textbf{4.7:Types of karma:} For ordinary people, karma is white/black/mixed. For the yogi, karma is neither:they act without binding consequences.
\item \textbf{4.8:} Karma ripens only when the right conditions of species, place, and time arise.
\item \textbf{4.9:Samskaras across lifetimes:} Memory and latent impressions share one form:unbroken continuity \emph{across births}. ``Deep habit patterns play out from lifetime to lifetime.'' (Dr. Ananda)
\item \textbf{4.10:} The will to live (Abhinivesha) is beginningless:confirming the reality of the cycle.
\item \textbf{4.12:} Past and future both exist in the present, differing only in characteristic:by knowing the present fully, one may know past and future.
\item \textbf{4.27:} Even at the highest level, deep samskaras can re-surface and stall progress:remain vigilant (Dr. Ananda).
\end{itemize}

\tiny{(Ref: Dr. Ananda Balayogi Bhavanani; [IT] I.K. Taimni; Nancy Wile)}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{4.29–4.34:Dharma Megha Samadhi and Kaivalya}

\begin{sanskrit}
प्रसंख्यानेऽप्यकुसीदस्य सर्वथा विवेकख्यातेर्धर्ममेघः समाधिः ॥४.२९॥
ततः क्लेशकर्मनिवृत्तिः ॥४.३०॥
\end{sanskrit}

\textbf{4.29:Dharma Megha Samadhi (धर्ममेघसमाधि):}
\begin{itemize}
\item ``The rain cloud of virtue'':even attachment to discriminative knowledge is released; the mind pours forth righteousness naturally
\item ``The torrential rainfall from this rain cloud of the highest nature washes away all arrogant impurities.'':Dr. Ananda
\end{itemize}

\textbf{4.30:} All kleshas and karma removed once and for all:the final liberation. ``At this point, Patanjali implies we become the Divine itself in the state of Kaivalya.'' (Dr. Ananda)

\begin{sanskrit}
पुरुषार्थशून्यानां गुणानां प्रतिप्रसवः कैवल्यं स्वरूपप्रतिष्ठा वा चितिशक्तिरिति ॥४.३४॥
\end{sanskrit}

\textbf{4.34:Kaivalya:} The gunas resolve back into their cause. Pure consciousness is established in its own true nature. The evolutionary journey ends. \textbf{Sat-Chit-Ananda}. ``We lose our sense of individuality in order to gain the sense of absolute universality.'' (Dr. Ananda)

\tiny{(Ref: Dr. Ananda Balayogi Bhavanani; [HA] Hariharananda Aranya; [IT] I.K. Taimni)}

\end{frame}

\begin{frame}[fragile]\frametitle{Philosophical Roots:A Universal Science}

\begin{itemize}
\item \textbf{Samkhya-Yoga:} Samkhya = \emph{theory} (why we suffer); Yoga = \emph{practice} (how to be free). The Bhagavad Gita is built on this Samkhya-Yoga foundation.
\item \textbf{Buddhism parallel:} Ashtanga Yoga's eight limbs parallel the Buddhist \emph{Noble Eightfold Path}. Brahmaviharas (1.33: Maitri, Karuna, Mudita, Upeksha) appear in both traditions:a universal human wisdom.
\item \textbf{Jainism parallel:} The five Yamas closely resemble the five major vows of Jainism. Ahimsa (non-violence) appears first in both.
\item \textbf{Modern psychology:} Klesha theory parallels cognitive distortions in CBT; Pratipaksha Bhavana = reframing; the five Chitta states = attention research.
\end{itemize}

\begin{center}
{\small ``The Yoga Sutras clarify esoteric concepts that are common to \emph{all traditions of Indian thought}.'' (Nancy Wile)\\
\textbf{Yoga does not belong to any religion. It belongs to humanity.}}
\end{center}

\tiny{(Ref: Wikipedia:Philosophical Roots; Nancy Wile; Dr. Ananda Balayogi Bhavanani)}

\end{frame}

\begin{frame}[fragile]\frametitle{Start Today:Your Personal Sadhana}

\begin{center}
{\normalsize \textbf{You do not need to master 196 sutras to begin. Start with one.}}
\end{center}

\vspace{3mm}
\textbf{A simple daily practice (10–15 min):}
\begin{enumerate}
\item \textbf{Morning:} Choose one method from 1.34–1.39 that feels natural:pranayama, inner light, or contemplating what you love most
\item \textbf{One sutra a day:} Read one sutra, sit quietly with it before moving to the next. ``Get into a quiet posture and notice what meaning you get from each one.'':Nancy Wile
\item \textbf{In relationships:} Practice Maitri-Karuna-Mudita-Upeksha in one interaction daily (1.33)
\item \textbf{When negative thought arises:} Pratipaksha Bhavana:say STOP, then cultivate the opposite (2.33)
\item \textbf{Evening:} Ask: ``Were my thoughts today facts or colored concepts?'' (1.5-1.6)
\end{enumerate}

\vspace{2mm}
\begin{center}
{\small ``When that practice is done for a long time, without a break, and with sincere devotion, then it becomes a firmly rooted, stable and solid foundation.'':1.14}
\end{center}
