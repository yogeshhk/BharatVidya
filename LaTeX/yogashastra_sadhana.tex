\begin{frame}[fragile]\frametitle{सकाळची साधना (ईशा $\sim$ ४५ मिनिटे)}

	\begin{itemize}
	\item \textbf{आसन (५--७ मिनिटे):}
		\begin{itemize}
		\item तितली आसन (Butterfly), बाळ पाळणा आसन (Baby Cradle), मांजर-गाय आसन (Cat-Cow)
		\end{itemize}
	\item \textbf{प्राणायाम + बंध (१५ मिनिटे):}
		\begin{itemize}
		\item अनुलोम-विलोम --- २ मिनिटे
		\item ॐकार मंत्र --- २१ वेळा
		\item जलद उथळ श्वास --- १२० वेळा
		\item तीन बंधांसह श्वास घेणे; तीन बंधांसह श्वास सोडणे
		\end{itemize}
	\item \textbf{ध्यान / स्थिरता (६--७ मिनिटे)}
	\end{itemize}

{\tiny (Ref: ईशा योग)}

\end{frame}

\begin{frame}[fragile]\frametitle{दिवसाची साधना}

	\begin{itemize}
	\item जागे झाल्यावर लगेच कोमट पाणी --- पचनक्रिया सुरू होते, विष काढून टाकते
	\item प्राण लिंग मुद्रा --- ५ मिनिटे (जीवनशक्ती वाढवते)
	\item शक्तिशाली टाळी --- २ मिनिटे (रक्तसंचार वाढवतो, हाताच्या नाड्या उत्तेजित करतो)
	\item आठवड्यातून २ दिवस: डिजिटल + अन्नाचे उपवास
	\item निष्क्रिय/प्रतीक्षेच्या क्षणांमध्ये \textbf{आनापानसती} --- श्वासाला मौनपणे आत-बाहेर पाहणे
	\item \textbf{कुंभक सराव:} १२ श्वास-स्तर $\times$ १२ श्वास = एकूण १४४ श्वास
		\begin{itemize}
		\item खोल श्वास घ्या, पोट फुगवा, ३ सेकंद धरा, नाकाने सोडा
		\end{itemize}
	\end{itemize}

{\tiny (Ref: वैयक्तिक साधना)}

\end{frame}

\begin{frame}[fragile]\frametitle{संध्याकाळची साधना (श्रीमत $\sim$ ६०--७० मिनिटे)}

	\begin{itemize}
	\item भ्रामरी प्राणायाम --- ६--७ मिनिटे
	\item अंतर मौन (Inner Silence) --- २० मिनिटे
	\item योग निद्रा --- ३० मिनिटे
	\item बासरी सराव (लांब स्वर) + धोरण/अंतर्दृष्टी पुस्तक वाचन --- २० मिनिटे
	\end{itemize}

\begin{frame}[fragile]\frametitle{रात्रीची शांतता + मूळ संकल्प}

	\begin{itemize}
	\item नाकपुड्यात + पायाच्या तळव्यात तूप लावणे
	\item ``ॐ श्री मात्रे नमः'' --- हळूवारपणे झोप येईपर्यंत जप
	\item ७ तास खोल झोप; लवकर जाग आल्यास राम नाम ६३ वेळा (पोट आकुंचित करून)
	\end{itemize}

	\vspace{3mm}
	\textbf{मूळ संकल्प:}
	\begin{itemize}
	\item चांगली बुद्धी (स्पष्टता) $\cdot$ चांगली शक्ती (बळ) $\cdot$ चांगले स्वरूप (उपस्थिति)
	\item शक्ती $\rightarrow$ शांती $\rightarrow$ ईश्वर
	\item साक्षी $\rightarrow$ अनासक्ती $\rightarrow$ ईश्वर
	\end{itemize}
